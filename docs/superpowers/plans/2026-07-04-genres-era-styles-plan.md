// Готовые стили — библиотека 15 авторов (styles.js). Клик пишет имя автора в
// Ориентиры стиля и правила стиля в Правила автора через уже существующий
// addRule() (дедуп встроен, см. state.js). Подсветка «подходит жанру» —
// сортировка/бейдж, не фильтрация: все 15 всегда видны и кликабельны.
function renderStylePresets(s){
  const genre = s.project.genre;
  const applied = new Set(s.style.refs||[]);
  const sorted = [...AUTHOR_STYLES].sort((a,b)=>
    (styleMatchesGenre(b,genre)?1:0) - (styleMatchesGenre(a,genre)?1:0));
  return `<div class="field" style="margin-top:22px;border-top:1px solid var(--border);padding-top:16px">
    <label>Готовые стили <span class="hint">(классика и жанровые авторы — добавляют имя в «Ориентиры» и набор правил в «Правила автора»)</span></label>
    <div style="display:grid;grid-template-columns:repeat(auto-fill,minmax(220px,1fr));gap:8px;margin-top:8px">
      ${sorted.map(st=>{
        const isApplied = applied.has(st.name);
        const fits = styleMatchesGenre(st, genre);
        return `<div class="card" style="padding:8px 10px">
          <div style="font-weight:600;font-size:13px">${esc(st.name)}${fits?' <span class="muted" style="font-weight:400;font-size:11px">· подходит жанру</span>':''}</div>
          <div class="muted" style="font-size:12px;margin:4px 0 8px">${esc(st.blurb)}</div>
          <button class="btn style-add" data-id="${st.id}" ${isApplied?'disabled':''} style="font-size:11px;padding:3px 9px">${isApplied?'✓ Добавлено':'+ Добавить'}</button>
        </div>`;
      }).join('')}
    </div>
  </div>`;
}

## Порядок выполнения

Задачи **1 → 2 → 3** должны идти последовательно (все три редактируют один и тот же файл `genres.js`, вставки в одни и те же функции — параллельное выполнение вызовет конфликты). Задача **4** зависит от завершения 1-3 (тот же файл).

Задача **5** (новый файл `styles.js`) не конфликтует по файлу ни с одной из 1-4, но имеет **содержательную (не файловую) зависимость от Задачи 1**: `AUTHOR_STYLES` в Задаче 5 использует в `genres[]` ровно те 4 новых значения жанра, которые Задача 1 добавляет в `GENRES` (`'ироничный детектив'`, `'ироничное фэнтези'`, `'сказка'`, `'юмористическая проза'`). Если Задачу 5 выполнить ДО Задачи 1 (например, распараллелив по признаку «разные файлы»), ошибки не будет — `styleMatchesGenre()` просто молча никогда не подсветит Пратчетта/Адамса для «ироничное фэнтези», Чехова/Довлатова для «ироничный детектив» и т.д., пока `GENRES` не пополнится. Поэтому зависимость 5-от-1 — по данным, а не по файлу, и её нельзя снять параллелизацией: **Задача 5 не должна начинаться, пока Задача 1 не завершена и не закоммичена**, даже если в моменте это выглядит как «независимая» работа с новым файлом.

Задача **6** зависит от завершения Задачи 5 (импортирует `styles.js`) и Задачи 4 (редактирует тот же файл `stages.js` — конфликт слияния при не строго последовательном выполнении). Порядок: **1 → 2 → 3 → 4 → 5 → 6 → 7**, полностью последовательно. Задача **7** — после всех остальных.
